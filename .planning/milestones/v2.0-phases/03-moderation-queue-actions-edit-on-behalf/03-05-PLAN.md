<edit_on_behalf_contract>
Per REQUIREMENTS.md MOD-14 + CONTEXT.md D-12 + RESEARCH.md §"7. D-12 status-after-save" + PATTERNS.md §H:

**Endpoint:** `PUT /api/moderation/listings/:id` (mounted under the router that's already at `/api/moderation`, so resolves to `/api/moderation/listings/:id`)

**Body:** Multipart form-data identical to PUT /api/properties/:id (Phase 2 D-22). Optional fields: title, description, address, price, propertyType, listingType, rooms, maxGuests, amenities, ... (mirror existing Property schema). Plus: optional `images` files array; optional `reason` (free-text, stored as `reasonNote` on the audit row).

**Mass-assignment-safe whitelist (PATTERNS §H):** Strip from updateData before the DB write:
- `ownerUid` (NEVER mutated — MOD-14 hard rule)
- `_id`, `id` (immutable)
- `createdAt` (immutable timestamp)
- `status` (forced by handler to 'live' regardless of body)
- `submittedAt` (Phase 2 owns)

**Status semantics (D-12 sub-decision):** Force `status: 'live'`, set `approvedAt: now`, `approvedByUid: req.firebaseUid`. Clear all rejection metadata (`rejectionReasonCode: null`, `rejectionReasonNote: null`, `rejectedAt: null`, `rejectedByUid: null`).

**Race-safety:** `findOneAndUpdate({_id, status: {$in: ['pending', 'rejected']}}, ...)` — edit-on-behalf is allowed on pending OR rejected (NOT on already-live; that's a post-edit owner workflow). Loser receives 409 ALREADY_MODERATED.

**Audit log:**
- `action: 'edit-on-behalf'`
- `actorUid: req.firebaseUid` (NEVER from body)
- `before`: changed-fields-only diff snapshot of original values
- `after`: changed-fields-only diff snapshot of new values + the forced status: 'live' flip
- `reasonNote`: req.body.reason || null (the free-text mod note for the audit trail)
- `at`: now

**Response:** 200 with the updated Property document (the same shape PropertyService.editAsModerator expects); 409 on race; 400 on validation error; 404 if listing does not exist (the findOneAndUpdate will return null in this case too — caller can't distinguish from 409, that's intentional and acceptable per UX)

const upload = require('../middleware/uploadS3');     // verify the existing multer-s3 middleware import path; if differently named, use whatever propertyRoutes.js uses

router.put('/listings/:id', upload.array('images', 40), async (req, res) => {
  try {
    const propertyId = req.params.id;

    // STEP 1: Build updateData from the request body, applying the mass-assignment whitelist.
    // PATTERNS §H — strip ownerUid, _id, id, createdAt, status, submittedAt before any DB write.
    const updateData = { ...(req.body || {}) };
    delete updateData.ownerUid;
    delete updateData._id;
    delete updateData.id;
    delete updateData.createdAt;
    delete updateData.status;
    delete updateData.submittedAt;
    delete updateData.approvedAt;
    delete updateData.approvedByUid;
    delete updateData.rejectedAt;
    delete updateData.rejectedByUid;
    delete updateData.rejectionReasonCode;
    delete updateData.rejectionReasonNote;
    delete updateData.actorUid;     // defensive — never accept actorUid from body
    delete updateData.firebaseUid;  // defensive — never accept firebaseUid from body

    // The mod's free-text reason for the audit log is in req.body.reason; remove it from updateData
    // so it doesn't accidentally become a Property field.
    const reasonForAudit = (typeof req.body.reason === 'string' && req.body.reason.trim()) ? req.body.reason.trim() : null;
    delete updateData.reason;

    // (Optional: handle image uploads — if req.files is non-empty, set updateData.images per the
    // existing propertyRoutes.js PUT pipeline. Mirror that block; this task does NOT need to invent
    // a new image-upload mechanism.)

    // STEP 2: Load the original document for the audit-log diff (one query before save).
    const original = await Property.findById(propertyId).lean();
    if (!original) {
      return res.status(404).json({ message: 'Property not found' });
    }

    // STEP 3: Compute the diff for the audit log (changed-fields-only).
    const before = {};
    const after = {};
    for (const k of Object.keys(updateData)) {
      const orig = original[k];
      const next = updateData[k];
      if (JSON.stringify(orig) !== JSON.stringify(next)) {
        before[k] = orig;
        after[k] = next;
      }
    }

    // STEP 4: Force status to 'live' (D-12 sub-decision). Always include the status flip in the diff.
    const now = new Date();
    before.status = original.status;
    after.status = 'live';
    updateData.status = 'live';
    updateData.approvedAt = now;
    updateData.approvedByUid = req.firebaseUid;
    // Clear rejection metadata since the mod has FIXED the issue
    updateData.rejectionReasonCode = null;
    updateData.rejectionReasonNote = null;
    updateData.rejectedAt = null;
    updateData.rejectedByUid = null;

    // STEP 5: Race-safe atomic update — allowed on pending OR rejected; not on live/archived.
    const result = await Property.findOneAndUpdate(
      { _id: propertyId, status: { $in: ['pending', 'rejected'] } },
      { $set: updateData },
      { new: true }
    );

    if (!result) {
      return res.status(409).json({
        code: 'ALREADY_MODERATED',
        message: 'This listing was already reviewed by another moderator.',
      });
    }

    // STEP 6: Audit-log follow-up insert (orphan-tolerant per D-10).
    try {
      await ModerationLog.create({
        actorUid: req.firebaseUid,
        action: 'edit-on-behalf',
        targetType: 'property',
        targetId: result._id,
        before,
        after,
        reasonCode: null,
        reasonNote: reasonForAudit,
        at: now,
      });
    } catch (auditErr) {
      console.error(JSON.stringify({
        evt: 'moderation_audit_orphan',
        action: 'edit-on-behalf',
        targetId: String(result._id),
        actorUid: req.firebaseUid,
        err: auditErr.message,
        ts: new Date().toISOString(),
      }));
    }

    return res.json(result);
  } catch (err) {
    console.error('Error in edit-on-behalf:', err);
    return res.status(500).json({ message: err.message });
  }
});
