        const now = new Date();

        // Q5 recommendation: capture pre-update status for clean audit diff (read overhead negligible).
        const beforeDoc = await Property.findById(req.params.id).lean();

        const result = await Property.findOneAndUpdate(
          { _id: req.params.id, ownerUid: req.firebaseUid, status: { $ne: 'archived' } },
          { $set: {
              status: 'archived',
              archivedAt: now,
              archivedByUid: req.firebaseUid,
              archivedReasonCode: null,        // owner-archive is no-reason per ARCH-01 + D-01
              archivedReasonNote: null,
            } },
          { new: true }
        );

        if (!result) {
          // Q4 recommendation: disambiguate 403/404/409 via follow-up findById read.
          const property = beforeDoc || await Property.findById(req.params.id);
          if (!property) return res.status(404).json({ message: 'Property not found' });
          if (property.ownerUid !== req.firebaseUid) {
            return res.status(403).json({ message: 'Access denied. You can only archive your own listings.' });
          }
          return res.status(409).json({ code: 'ALREADY_ARCHIVED', message: 'This listing is already archived.' });
        }

        try {
          await ModerationLog.create({
            actorUid: req.firebaseUid,                     // PATTERN §E — JWKS sub, NEVER body
            action: 'archive',                              // per Plan 01 D-05 enum extension
            targetType: 'property',
            targetId: result._id,
            before: { status: beforeDoc ? beforeDoc.status : 'unknown' },
            after: { status: 'archived' },
            reasonCode: null,
            reasonNote: null,
            at: now,
          });
        } catch (auditErr) {
          // Per D-18: status flip is NOT rolled back; audit is forward-fit observability.
          console.error(JSON.stringify({

            action: 'archive',
            targetId: String(result._id),
            actorUid: req.firebaseUid,
            err: auditErr.message,
            ts: new Date().toISOString(),
          }));
        }

        return res.json(result);
      } catch (err) {
        console.error('Error owner-archiving listing:', err);
        return res.status(500).json({ message: err.message });
      }
    });
