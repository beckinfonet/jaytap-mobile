    // (valid | expired | tampered) × (admin | moderator | user)
    // Source: PITFALLS.md Pitfall 2 + RESEARCH.md Validation Architecture (lines 1339-1363).
    //
    // Generates an in-memory RS256 keypair and signs tokens with controlled iss/aud/iat/exp/sub claims.
    // The JWKS endpoint is mocked at the test level — the test file replaces createRemoteJWKSet
    // with a function that returns the test public key, so jose verifies against this keypair.

    const { SignJWT, generateKeyPair, exportJWK } = require('jose');

    const TEST_FIREBASE_PROJECT_ID = 'jaytap-test-project';
    const TEST_ISSUER = `https://securetoken.google.com/${TEST_FIREBASE_PROJECT_ID}`;

    let _keys = null;
    async function getKeys() {
      if (!_keys) {
        const kp = await generateKeyPair('RS256');
        const publicJwk = await exportJWK(kp.publicKey);
        publicJwk.kid = 'test-kid-1';
        publicJwk.alg = 'RS256';
        publicJwk.use = 'sig';
        _keys = { privateKey: kp.privateKey, publicJwk };
      }
      return _keys;
    }

    /**
     * Build a token with controlled claims.
     * @param {Object} opts
     * @param {string} opts.sub - Firebase uid (the user's localId)
     * @param {string} opts.role - 'user' | 'moderator' | 'admin' (used by tests to look up the corresponding Mongo record)
     * @param {string} [opts.kind='valid'] - 'valid' | 'expired' | 'tampered'
     * @param {string} [opts.audience=TEST_FIREBASE_PROJECT_ID]
     * @param {string} [opts.issuer=TEST_ISSUER]
     */
    async function buildToken({ sub, role, kind = 'valid', audience = TEST_FIREBASE_PROJECT_ID, issuer = TEST_ISSUER }) {
      const { privateKey } = await getKeys();
      const nowSec = Math.floor(Date.now() / 1000);
      const exp = kind === 'expired' ? nowSec - 60 : nowSec + 3600;
      const iat = kind === 'expired' ? nowSec - 3660 : nowSec;

      const token = await new SignJWT({ sub, role })
        .setProtectedHeader({ alg: 'RS256', kid: 'test-kid-1' })
        .setIssuedAt(iat)
        .setIssuer(issuer)
        .setAudience(audience)
        .setSubject(sub)
        .setExpirationTime(exp)
        .sign(privateKey);

      if (kind === 'tampered') {
        // Flip one byte of the signature (last segment after the second '.')
        const parts = token.split('.');
        const sig = Buffer.from(parts[2], 'base64url');
        sig[0] ^= 0x01;
        parts[2] = sig.toString('base64url');
        return parts.join('.');
      }
      return token;
    }

    module.exports = {
      TEST_FIREBASE_PROJECT_ID,
      TEST_ISSUER,
      getKeys,
      buildToken,
    };
