    ```
    // Phase 11 GEO-01 — Forward geocode (address → coords + displayName).
    //
    // Body: { address: string (required), citySlug?: string, lang?: 'en'|'ru' }
    // Response 200: { lat: number, lng: number, displayName: string }
    // Response 400: { code: 'ADDRESS_REQUIRED' } — missing or whitespace-only address
    // Response 404: { code: 'NOT_FOUND' } — geocoder returned null (no result / timeout / network error)
    //
    // citySlug → server-side City lookup → viewbox bias (anti-"random pin" defense per
    // 11-CONTEXT.md <specifics> "THREE layers"). Unknown citySlug degrades gracefully
    // (warn + proceed without bias) — the geocode still scopes to KG/KZ/UZ via
    // countrycodes in the helper.
    //
    // Auth gate: verifyFirebaseToken + requireListingCapability — same as POST /cities
    // (mirrors the spam-guard pattern; per-uid rate limiting deferred to M6+).
    router.post('/geocode', verifyFirebaseToken, requireListingCapability, async (req, res) => {
      try {
        const { address, citySlug, lang } = req.body;
        if (!address || typeof address !== 'string' || !address.trim()) {
          return res.status(400).json({ code: 'ADDRESS_REQUIRED' });
        }
        const sanitizedLang = (lang === 'ru' || lang === 'en') ? lang : 'en';

        let bias;
        if (citySlug && typeof citySlug === 'string') {
          const city = await City.findOne({ slug: citySlug, status: 'approved' }).lean();
          if (city && city.centroid && typeof city.centroid.lat === 'number' && typeof city.centroid.lng === 'number') {
            bias = { lat: city.centroid.lat, lng: city.centroid.lng };
          } else {
            console.warn('geocode: citySlug not found or has no centroid; proceeding without viewbox bias:', citySlug);
          }
        }

        const result = await geocodeAddress(address, { bias, lang: sanitizedLang });
        if (!result) {
          console.warn('geocode: helper returned null for address:', address.slice(0, 200));
          return res.status(404).json({ code: 'NOT_FOUND' });
        }
        res.json({
          lat: result.latitude,
          lng: result.longitude,
          displayName: result.displayName ?? '',
        });
      } catch (err) {
        console.error('Error in /api/locations/geocode:', err);
        res.status(500).json({ message: err.message });
      }
    });
    ```

    STEP 3 — Append POST /reverse-geocode route (after /geocode, still before `module.exports`):

    ```
    // Phase 11 GEO-02 — Reverse geocode (coords → displayName).
    //
    // Body: { lat: number (required, finite), lng: number (required, finite), lang?: 'en'|'ru' }
    // Response 200: { displayName: string }
    // Response 400: { code: 'COORDS_REQUIRED' } — lat/lng missing or non-finite
    // Response 404: { code: 'NOT_FOUND' } — reverseGeocode returned null
    //
    // Called by Step2Location.tsx on pin drop/drag — best-effort fill of an empty
    // location.address field. Same auth gate as /geocode.
    router.post('/reverse-geocode', verifyFirebaseToken, requireListingCapability, async (req, res) => {
      try {
        const { lat, lng, lang } = req.body;
        if (typeof lat !== 'number' || !Number.isFinite(lat) || typeof lng !== 'number' || !Number.isFinite(lng)) {
          return res.status(400).json({ code: 'COORDS_REQUIRED' });
        }
        const sanitizedLang = (lang === 'ru' || lang === 'en') ? lang : 'en';

        const result = await reverseGeocode({ lat, lng, lang: sanitizedLang });
        if (!result) {
          console.warn('reverse-geocode: helper returned null for coords:', lat, lng);
          return res.status(404).json({ code: 'NOT_FOUND' });
        }
        res.json({ displayName: result.displayName ?? '' });
      } catch (err) {
        console.error('Error in /api/locations/reverse-geocode:', err);
        res.status(500).json({ message: err.message });
      }
    });
    ```
