    const express = require('express');
    const City = require('../models/City');
    const District = require('../models/District');
    const { verifyFirebaseToken, optionalAuth } = require('../middleware/verifyFirebaseToken');
    const { requireListingCapability } = require('../middleware/requireListingCapability');
    const { slugify } = require('transliteration');

    const router = express.Router();

    // GET /api/locations/cities — returns approved + caller's own pending
    router.get('/cities', optionalAuth, async (req, res) => {
      const callerUid = req.firebaseUid; // may be undefined for anon
      const cities = await City.find({
        $or: [
          { status: 'approved' },
          ...(callerUid ? [{ status: 'pending', createdByUid: callerUid }] : []),
        ],
      }).sort({ country: 1, 'label.en': 1 }).lean();
      res.json({ cities });
    });

    // POST /api/locations/cities — auth required + listing capability; creates pending
    router.post('/cities', verifyFirebaseToken, requireListingCapability, async (req, res) => {
      const { label, country, centroid } = req.body;
      if (!label?.ru || !label?.en) return res.status(400).json({ code: 'LABEL_REQUIRED' });
      if (!['KG', 'KZ', 'UZ'].includes(country)) return res.status(400).json({ code: 'INVALID_COUNTRY' });
      // Per Tradeoff §F — centroid required from client (drop-pin-first UX)
      if (!centroid?.lat || !centroid?.lng) return res.status(400).json({ code: 'CENTROID_REQUIRED' });

      const slug = slugify(label.en || label.ru, { lowercase: true });

    Skeleton:
    ```js
    #!/usr/bin/env node
    /**
     * Phase 2 Plan 01 — Operator-supervised Location dictionary seed.
     * Pattern source: src/scripts/migrate-listings-m3.js (M3 P1 D-02).
     * Memory `backend-node-version.md`: requires Node >= 22.12 (`nvm use 24`).
     * Memory `geographic-scope.md`: KG/KZ/UZ launch scope.
     *
     * Usage:
     *   nvm use 24
     *   npm run seed:locations -- --dry-run     # Print what WOULD be inserted; no DB writes
     *   npm run seed:locations                  # Live insert
     *   npm run seed:locations -- --verify=PASS # Post-run verification check
     *
     * Exit codes:
     *   0 = success / verify PASS
     *   1 = seed error (DB unreachable, schema validation fail, etc.)
     *   2 = operator misuse (--verify without =PASS, Node version too old)
     */

    require('dotenv').config();
    const path = require('path');
    const fs = require('fs');
    const mongoose = require('mongoose');

    if (Number(process.versions.node.split('.')[0]) < 22) {
      console.error('FATAL: Node ≥22.12 required (run `nvm use 24` first).');
      process.exit(2);
    }

    const args = process.argv.slice(2);
    const isDryRun = args.includes('--dry-run');
    const isVerify = args.includes('--verify=PASS');
    if (args.includes('--verify') && !args.includes('--verify=PASS')) {
      console.error('FATAL: `--verify` requires the `=PASS` suffix to confirm intent.');
      process.exit(2);
    }

    const City = require('../models/City');
    const District = require('../models/District');

    const dataPath = path.join(__dirname, 'seed-locations-m3-data.json');
    const data = JSON.parse(fs.readFileSync(dataPath, 'utf8'));

    async function main() {
      const mongoUri = process.env.MONGODB_URI;
      if (!mongoUri) { console.error('FATAL: MONGODB_URI env not set.'); process.exit(1); }
      await mongoose.connect(mongoUri);
      console.log(`Connected to MongoDB. Mode: ${isDryRun ? 'DRY-RUN' : isVerify ? 'VERIFY' : 'LIVE'}`);

      if (isVerify) {
        const cityCount = await City.countDocuments({ status: 'approved' });
        const districtCount = await District.countDocuments({ status: 'approved' });
        console.log(`Approved cities: ${cityCount}`);
        console.log(`Approved districts: ${districtCount}`);
        if (cityCount === 0 || districtCount === 0) {
          console.error('FATAL: VERIFY failed — at least one approved city + district required.');
          process.exit(1);
        }
        console.log('VERIFY PASS.');
        process.exit(0);
      }

      let citiesInserted = 0;
      let citiesSkipped = 0;
      const cityIdBySlug = new Map();

      for (const c of data.cities) {
        const existing = await City.findOne({ slug: c.slug });
        if (existing) {
          cityIdBySlug.set(c.slug, existing._id);
          citiesSkipped++;
          continue;
        }
        if (isDryRun) {
          console.log(`[DRY-RUN] WOULD insert city: ${c.slug} (${c.label.en})`);
          citiesInserted++;
          continue;
        }
        const created = await City.create({
          slug: c.slug,
          label: c.label,
          country: c.country,
          centroid: c.centroid,
          status: 'approved',
          createdByUid: 'seed-script',
          approvedByUid: 'seed-script',
          approvedAt: new Date(),
        });
        cityIdBySlug.set(c.slug, created._id);
        citiesInserted++;
      }

      let districtsInserted = 0;
      let districtsSkipped = 0;
      for (const d of data.districts) {
        const cityId = cityIdBySlug.get(d.citySlug);
        if (!cityId) {
          console.warn(`WARN: district ${d.slug} references unknown city ${d.citySlug} — skipping`);
          continue;
        }
        const existing = await District.findOne({ cityId, slug: d.slug });
        if (existing) { districtsSkipped++; continue; }
        if (isDryRun) {
          console.log(`[DRY-RUN] WOULD insert district: ${d.citySlug}/${d.slug} (${d.label.en})`);
          districtsInserted++;
          continue;
        }
        await District.create({
          cityId,
          slug: d.slug,
          label: d.label,
          status: 'approved',
          createdByUid: 'seed-script',
          approvedByUid: 'seed-script',
          approvedAt: new Date(),
        });
        districtsInserted++;
      }

      console.log(`Cities: ${citiesInserted} inserted, ${citiesSkipped} skipped (already exist)`);
      console.log(`Districts: ${districtsInserted} inserted, ${districtsSkipped} skipped`);
      if (isDryRun) console.log('Dry run complete. Re-run without --dry-run to apply.');
      else console.log('Seed complete. Run with --verify=PASS to confirm.');
      await mongoose.disconnect();
      process.exit(0);
    }

    main().catch(err => { console.error('FATAL:', err); mongoose.disconnect().catch(()=>{}); process.exit(1); });
    ```
