Orphan-mark handles unrepairable cases (no User with that phone, or 2+ ambiguous matches).

    const mongoose = require('mongoose');
    const dotenv = require('dotenv');
    const LandlordApplication = require('../models/LandlordApplication');
    const LandlordApplicationAuditLog = require('../models/LandlordApplicationAuditLog');

    /**
     * Pure classifier — exported for unit testing.
     * Caller MUST do the application.uid → User lookup BEFORE invoking; classifyRow is only called for misses.
     * @param {object} application — LandlordApplication doc with .uid + .phone
     * @param {Array<{uid: string, phone?: string}>} candidateUsers — Users matching application.phone
     * @returns {{kind: 'phone-match-flip', newUid: string} | {kind: 'orphan-mark', reason: 'no-phone-match' | 'ambiguous-phone-match'}}
     */
    function classifyRow(application, candidateUsers) {
      const otherMatches = candidateUsers.filter(u => u.uid !== application.uid);
      if (otherMatches.length === 1) {
        return { kind: 'phone-match-flip', newUid: otherMatches[0].uid };
      }
      return {
        kind: 'orphan-mark',
        reason: otherMatches.length === 0 ? 'no-phone-match' : 'ambiguous-phone-match',
      };
    }

    async function alreadyFlippedByMigration(applicationId, currentUid) {
      const prior = await LandlordApplicationAuditLog.findOne({
        applicationId,
        action: 'uid-repair',
        'after.uid': currentUid,
      });
      return prior !== null;
    }

    async function processApplication(app, opts = { dryRun: false }) {
      // Idempotency check — skip if already orphaned or already flipped to current uid
      if (app.status === 'orphaned') {
        return { decision: 'skip-already-orphaned' };
      }
      if (await alreadyFlippedByMigration(app._id, app.uid)) {
        return { decision: 'skip-already-flipped' };
      }

      // Hit-skip: application.uid resolves to a real User → leave alone
      const owner = await User.findOne({ uid: app.uid });
      if (owner) {
        return { decision: 'hit-skip' };
      }

      // Phone-match attempt
      const candidates = await User.find({ phone: app.phone });
      const decision = classifyRow(app, candidates);

      if (opts.dryRun) {
        return { decision: `dry-run:${decision.kind}`, detail: decision };
      }

      if (decision.kind === 'phone-match-flip') {
        await LandlordApplication.updateOne(
          { _id: app._id },
          { $set: { uid: decision.newUid } },
        );
        await LandlordApplicationAuditLog.create({
          applicationId: app._id,
          applicantUid: decision.newUid,
          actorUid: 'system-migration',
          action: 'uid-repair',
          before: { uid: app.uid },
          after: { uid: decision.newUid },
          at: new Date(),
        });
        return { decision: 'phone-match-flip', detail: decision };
      }

      // orphan-mark
      await LandlordApplication.updateOne(
        { _id: app._id },
        { $set: { status: 'orphaned' } },
      );
      await LandlordApplicationAuditLog.create({
        applicationId: app._id,
        applicantUid: app.uid,
        actorUid: 'system-migration',
        action: 'uid-orphan-mark',
        before: { status: app.status },
        after: { status: 'orphaned' },
        reasonNote: decision.reason,
        at: new Date(),
      });
      return { decision: 'orphan-mark', detail: decision };
    }

    async function main() {
