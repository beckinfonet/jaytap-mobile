**PATCH /users/:uid/role handler (per RESEARCH §Code Example 3 + Concurrency Risk Register Risk #1 rollback path):**

```js
// PATCH /api/admin/users/:uid/role  body: {userType: 'user'|'moderator'|'admin'}
// ADMIN-02 + ADMIN-03 + ADMIN-04 + ADMIN-05 + ADMIN-07
// Order of operations:
//   1. SELF_MUTATION pre-check (Pattern 4 — cheapest reject path, O(1) string compare)
//   2. userType enum validation → 400 invalid
//   3. Lookup target user via {uid: targetUid} — NOT {firebaseUid:...} (User schema uses `uid`)
//   4. Same-tier no-op fast path: 200 + return doc unchanged (CONTEXT D-Discretion)
//   5. Demotion detection (Pitfall 3): only demotion bumps roleRevokedAt
//   6. LAST_ADMIN_LOCKOUT pre-flight count (Pitfall 1)
//   7. Race-safe atomic findOneAndUpdate({uid, userType: fromRole}) (Pattern D)
//      → null result = 409 ROLE_ALREADY_CHANGED
//   8. Post-write rollback close (Risk #1): if final admin count is 0,
//      restore old role and return 409 LAST_ADMIN_LOCKOUT.
//   9. Audit-row insert in own try/catch (Pattern E orphan-tolerant)
//      actorUid: req.firebaseUid ONLY — NEVER from req.body
//  10. Return 200 with full updated User doc minus __v
router.patch('/users/:uid/role', async (req, res) => {
  try {
    const targetUid = req.params.uid;
    const { userType: newRole } = req.body;

    // Step 1 — Pattern 4 self-mutation pre-check (D-11)
    if (req.firebaseUid === targetUid) {
      return res.status(403).json({
        code: 'SELF_MUTATION',
        message: 'Admins cannot change their own role.',
      });
    }

    // Step 2 — userType enum validation
    if (!VALID_USER_TYPES.includes(newRole)) {
      return res.status(400).json({
        message: `userType must be one of ${VALID_USER_TYPES.join(', ')}`,
      });
    }

    // Step 3 — Lookup target. User schema field is `uid` (NOT firebaseUid).
    const target = await User.findOne({ uid: targetUid }).lean();
    if (!target) {
      return res.status(404).json({
        code: 'USER_NOT_FOUND',
        message: 'User not found',
      });
    }
    const fromRole = target.userType;

    // Step 4 — Same-tier no-op (CONTEXT D-Discretion: 200 with no roleRevokedAt change).
    if (fromRole === newRole) {
      const fresh = await User.findOne({ uid: targetUid }).lean();
      const { __v, ...publicDoc } = fresh;
      return res.json(publicDoc);
    }

    // Step 5 — Pitfall 3: demotion detection. roleRevokedAt bumps on demotion ONLY.
    const isDemotion = (
      (fromRole === 'admin' && newRole !== 'admin') ||
      (fromRole === 'moderator' && newRole === 'user')
    );

    // Step 6 — Pitfall 1 last-admin pre-flight (only when demoting an admin).
    // Counts admins EXCLUDING the target → answers "after this demotion, will any admins remain?"
    if (fromRole === 'admin' && newRole !== 'admin') {
      const otherAdmins = await User.countDocuments({
        userType: 'admin',
        _id: { $ne: target._id },
      });
      if (otherAdmins < 1) {
        return res.status(409).json({
          code: 'LAST_ADMIN_LOCKOUT',
          message: 'Cannot change role — this user is the only admin. Promote another user to admin first.',
        });
      }
    }

    // Step 7 — Pattern D race-safe atomic. Filter on current userType — concurrent
    // role change makes findOneAndUpdate return null → 409 ROLE_ALREADY_CHANGED.
    const $set = { userType: newRole };
    if (isDemotion) $set.roleRevokedAt = new Date(); // ROLE-11 invariant; conditional bump per Pitfall 3.

    const result = await User.findOneAndUpdate(
      { uid: targetUid, userType: fromRole },
      { $set },
      { new: true }
    );

    if (!result) {
      return res.status(409).json({
        code: 'ROLE_ALREADY_CHANGED',
        message: "This user's role was already changed by another admin.",
      });
    }

    // Step 8 — Concurrency Risk #1 post-write rollback close.
    // If the only-other admin was concurrently demoted between our pre-flight and
    // our atomic update, the system is now zero-admin. Restore the old role.
    if (isDemotion && fromRole === 'admin') {
      const finalAdminCount = await User.countDocuments({ userType: 'admin' });
      if (finalAdminCount < 1) {
        await User.updateOne(
          { _id: result._id },
          { $set: { userType: fromRole } } // do NOT bump roleRevokedAt on rollback
        );
        console.error(JSON.stringify({
          evt: 'last_admin_lockout_rollback',
          targetUid, actorUid: req.firebaseUid,
          fromRole, attemptedToRole: newRole,
          ts: new Date().toISOString(),
        }));
        return res.status(409).json({
          code: 'LAST_ADMIN_LOCKOUT',
          message: 'Cannot change role — this user is the only admin. Promote another user to admin first.',
        });
      }
    }

    // Step 9 — Pattern E audit-row write (orphan-tolerant).
    // actorUid: req.firebaseUid — JWKS-verified sub, NEVER req.body (Pattern B anti-spoofing).
    try {
      await RoleChangeLog.create({
        actorUid: req.firebaseUid,
        targetUid: targetUid,
        fromRole,
        toRole: newRole,
        at: new Date(),
      });
    } catch (auditErr) {
      console.error(JSON.stringify({
        evt: 'role_change_audit_orphan',
        targetUid, actorUid: req.firebaseUid,
        fromRole, toRole: newRole,
        err: auditErr.message,
        ts: new Date().toISOString(),
      }));
    }

    // Step 10 — Return full updated User doc per CONTEXT D-Discretion. Strip __v.
    const { __v, ...publicDoc } = result.toObject();
    return res.json(publicDoc);
  } catch (err) {
    console.error('Error in PATCH /api/admin/users/:uid/role:', err);
    return res.status(500).json({ message: err.message });
  }
});

module.exports = { router, VALID_USER_TYPES };
```
